Usage Tips

  • Use text-wonderland-rose-petal for primary text highlights
  • Use bg-wonderland-sugar-pink for soft backgrounds
  • For themed components: bg-wonderland-dreamy-primary text-wonderland-dreamy-accent
  • For contrasting buttons: bg-wonderland-trio-vibrant-primary text-white
  • For subtle borders: border-wonderland-scone-brown

CSS Variables Reference

:root {
  /* Individual colors */
  --wonderland-rose-petal: #da62a0;
  --wonderland-pastel-pink: #edbbd5;
  --wonderland-sugar-pink: #f5dae8;
  --wonderland-mint-tea: #a5cdc2;
  --wonderland-clear-blue: #d3ecfb;
  --wonderland-baked-brown: #bc8c7f;
  --wonderland-scone-brown: #caae85;
  --wonderland-agate: #a00f20;
  
  /* Dreamy theme */
  --wonderland-dreamy-primary: var(--wonderland-pastel-pink);
  --wonderland-dreamy-secondary: var(--wonderland-rose-petal);
  --wonderland-dreamy-tertiary: var(--wonderland-sugar-pink);
  --wonderland-dreamy-accent: var(--wonderland-mint-tea);
  --wonderland-dreamy-neutral: var(--wonderland-scone-brown);
  
  /* Queen theme */
  --wonderland-queen-primary: var(--wonderland-rose-petal);
  --wonderland-queen-secondary: var(--wonderland-pastel-pink);
  --wonderland-queen-tertiary: var(--wonderland-clear-blue);
  --wonderland-queen-accent: var(--wonderland-baked-brown);
  --wonderland-queen-neutral: var(--wonderland-scone-brown);
  --wonderland-queen-highlight: var(--wonderland-agate);
}
